# Methods

## Example

```vue
<template>
  <umo-office-viewer ref="viewerRef" v-bind="options" />
  <button @click="getOptions">Get options</button>
  <button @click="openSidebar">Open sidebar</button>
  <button @click="downloadFile">Download</button>
</template>

<script setup>
import { ref } from 'vue'
import { UmoOfficeViewer } from '@umoteam/office-viewer'

const viewerRef = ref(null)
const options = ref({})

const getOptions = () => viewerRef.value.getOptions()
const openSidebar = () => viewerRef.value.openAside('thumbs')
const downloadFile = () => viewerRef.value.download()
</script>
```
